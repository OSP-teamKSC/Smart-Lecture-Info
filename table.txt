create table lecture (
code char(15) NOT NULL, 
name char(20) NOT NULL,
credit double,
theory int,
practice int,
type char(10),
max int NOT NULL,
time varchar(50),
lec_room char(20),
campus char(10) NOT NULL,
college char(25) NOT NULL,
major varchar(50) NOT NULL,
open_major char(15),
professor char(10),
team char(5),
smart_learning char(5),
overseas char(5),
ranks char(10),
pro_coll char(25),
pro_major varchar(50),
language char(10),
etc char(10),
primary key(code)
);
